js:
// Get the navbar and collapse elements
    const navbar = document.querySelector('.navbar');
    const collapse = document.getElementById('navbarSupportedContent');

    // Handle scroll: add/remove "scrolled" class based on scroll position or menu open
    function updateNavbar() {
      if (window.scrollY > 50 || collapse.classList.contains('show')) {
        navbar.classList.add('scrolled');
      } else {
        navbar.classList.remove('scrolled');
      }
    }

    window.addEventListener('scroll', updateNavbar);

    collapse.addEventListener('shown.bs.collapse', () => {
      navbar.classList.add('clicked'); 
    });

    collapse.addEventListener('hidden.bs.collapse', () => {
      navbar.classList.remove('clicked'); 
      updateNavbar(); 
    });

    updateNavbar();
